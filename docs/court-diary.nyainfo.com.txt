# sudo nano /etc/nginx/sites-available/court-diary.nyainfo.com

server {
    listen 443 ssl;
    server_name court-diary.nyainfo.com;

    ssl_certificate /etc/letsencrypt/live/nyainfo.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/nyainfo.com/privkey.pem;include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    root /home/myuser/working/courtease-ui-e1fa5329/dist;
    index index.html;

    # React UI
    location / {
        try_files $uri /index.html;
    }

    # Proxy API calls
    location /rest/ {
        proxy_pass http://127.0.0.1:9000/api/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_connect_timeout 120s;
        proxy_send_timeout 120s;
        proxy_read_timeout 120s;
        send_timeout 120s;
    }
}
